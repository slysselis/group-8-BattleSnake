import { floodFill } from './floodFill.js';

/**
 * Chooses the best move using flood fill algorithm to calculate available space
 * @param {Object} gameState - The current game state from Battlesnake API
 * @param {Object} gameState.you - Your snake's data
 * @param {Array} gameState.you.body - Array of coordinate objects representing snake body
 * @param {Object} gameState.board - The game board data
 * @param {number} gameState.board.width - Width of the game board
 * @param {number} gameState.board.height - Height of the game board
 * @param {Array} gameState.board.snakes - Array of all snakes on the board
 * @param {Object} isMoveSafe - Object indicating which moves are safe
 * @param {boolean} isMoveSafe.up - Whether moving up is safe
 * @param {boolean} isMoveSafe.down - Whether moving down is safe
 * @param {boolean} isMoveSafe.left - Whether moving left is safe
 * @param {boolean} isMoveSafe.right - Whether moving right is safe
 * @returns {Object} Object with direction keys and flood fill scores as values
 * @example
 * // Returns flood fill scores for each safe direction
 * const scores = chooseMoveWithFloodFill(gameState, isMoveSafe);
 * // Result: { up: 12, right: 3, left: 8 }
 */
function chooseMoveWithFloodFill(gameState, isMoveSafe) {
  const directions = ["up", "down", "left", "right"];
  const myHead = gameState.you.body[0];
  const width = gameState.board.width;
  const height = gameState.board.height;

  // Create empty board representation
  const board = Array.from({ length: height }, () => Array(width).fill('.'));

  // Block snake bodies on the board
  gameState.board.snakes.forEach(snake => {
    snake.body.forEach(segment => {
      board[segment.y][segment.x] = 'X';
    });
  });

  const scores = {};
  for (let dir of directions) {
    if (!isMoveSafe[dir]) continue;

    // Calculate new head position for this direction
    let newHead = { ...myHead };
    if (dir === "up") newHead.y += 1;
    if (dir === "down") newHead.y -= 1;
    if (dir === "left") newHead.x -= 1;
    if (dir === "right") newHead.x += 1;

    // Calculate flood fill score for this move
    scores[dir] = floodFill(board, newHead);
  }

  return scores;
}

export default chooseMoveWithFloodFill;
