/**
 * Called on every turn to determine your next move
 * Implements comprehensive move logic including collision avoidance, flood fill analysis, and food seeking
 * @param {Object} gameState - The current game state from Battlesnake API
 * @param {number} gameState.turn - Current turn number
 * @param {Object} gameState.you - Your snake's data
 * @param {Array} gameState.you.body - Array of coordinate objects representing snake body
 * @param {number} gameState.you.health - Current health (0-100)
 * @param {number} gameState.you.length - Current snake length
 * @param {Object} gameState.board - The game board data
 * @param {number} gameState.board.width - Width of the game board
 * @param {number} gameState.board.height - Height of the game board
 * @param {Array} gameState.board.food - Array of food coordinate objects
 * @param {Array} gameState.board.snakes - Array of all snakes on the board
 * @returns {Object} Move decision object
 * @property {string} move - The chosen direction ("up", "down", "left", or "right")
 * @example
 * // Make a move decision
 * const decision = move(gameState);
 * // Result: { move: "up" }
 */
function move(gameState) {
  console.log(`MOVE ${gameState.turn}: Beginning turn`);

  // Initialize all moves as safe
  let isMoveSafe = {
    up: true,
    down: true,
    left: true,
    right: true
  };

  // Apply all collision avoidance logic in order of importance
  isMoveSafe = preventBackwardsMovement(gameState, isMoveSafe);
  isMoveSafe = preventBorderCollision(gameState, isMoveSafe);
  isMoveSafe = preventSelfCollision(gameState, isMoveSafe);
  isMoveSafe = preventSnakeCollision(gameState, isMoveSafe);
  isMoveSafe = headToHead(gameState, isMoveSafe);
  isMoveSafe = allowTailMovement(gameState, isMoveSafe); // if needed

  // Get flood fill scores for each safe move to evaluate available space
  const floodScores = chooseMoveWithFloodFill(gameState, isMoveSafe);

  // Find the move with the most available space
  const bestFloodMove = Object.entries(floodScores)
    .sort((a, b) => b[1] - a[1]) // descending order by space available
    .map(entry => entry[0])[0];

  // Check if we should prioritize food
  const foodMove = moveTowardFood(gameState, isMoveSafe);

  // Prefer food if health is low or available space is limited
  if (gameState.you.health < 70 || (bestFloodMove && floodScores[bestFloodMove] < 10)) {
    if (foodMove) {
      console.log(`MOVE ${gameState.turn}: Preferring food → ${foodMove}`);
      return { move: foodMove };
    }
  }

  // Use flood fill move if available
  if (bestFloodMove) {
    console.log(`MOVE ${gameState.turn}: Flood fill → ${bestFloodMove}`);
    return { move: bestFloodMove };
  }

  // Fallback: pick any remaining safe move randomly
  const safeMoves = Object.keys(isMoveSafe).filter(key => isMoveSafe[key]);
  if (safeMoves.length === 0) {
    console.log(`MOVE ${gameState.turn}: No safe moves! Going down.`);
    return { move: "down" };
  }

  const randomMove = safeMoves[Math.floor(Math.random() * safeMoves.length)];
  console.log(`MOVE ${gameState.turn}: Random safe move → ${randomMove}`);
  return { move: randomMove };
}

// Start the Battlesnake server with our handler functions

  move: move,
  end: end
