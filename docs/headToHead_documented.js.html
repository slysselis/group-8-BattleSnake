/**
 * Prevents head-to-head collisions with enemy snakes of equal or greater length
 * @param {Object} gameState - The current game state from Battlesnake API
 * @param {Object} gameState.you - Your snake's data
 * @param {Array} gameState.you.body - Array of coordinate objects representing your snake body
 * @param {number} gameState.you.length - Length of your snake
 * @param {string} gameState.you.id - Unique identifier for your snake
 * @param {Object} gameState.board - The game board data
 * @param {Array} gameState.board.snakes - Array of all snakes on the board
 * @param {Object} isMoveSafe - Object indicating which moves are currently safe
 * @param {boolean} isMoveSafe.up - Whether moving up is safe
 * @param {boolean} isMoveSafe.down - Whether moving down is safe
 * @param {boolean} isMoveSafe.left - Whether moving left is safe
 * @param {boolean} isMoveSafe.right - Whether moving right is safe
 * @returns {Object} Updated isMoveSafe object with head-to-head collisions marked as unsafe
 * @example
 * // Avoid head-to-head with bigger snakes
 * const safeMoves = avoidHeadToHead(gameState, isMoveSafe);
 * // safeMoves.up might be false if enemy snake could move there
 */
function avoidHeadToHead(gameState, isMoveSafe) {
    const myHead = gameState.you.body[0];
    const myLength = gameState.you.length;
    const opponents = gameState.board.snakes.filter(snake => snake.id !== gameState.you.id);
  
    // Check each opponent snake
    opponents.forEach(snake => {
      const enemyHead = snake.body[0];
      const enemyLength = snake.length;
  
      // Predict where the enemy head might move next turn
      const enemyMoves = [
        { x: enemyHead.x + 1, y: enemyHead.y }, // right
        { x: enemyHead.x - 1, y: enemyHead.y }, // left
        { x: enemyHead.x, y: enemyHead.y + 1 }, // up
        { x: enemyHead.x, y: enemyHead.y - 1 }  // down
      ];
  
      // Avoid head-to-head collision with bigger or same-length snakes
      enemyMoves.forEach(move => {
        if (enemyLength >= myLength) { // Only avoid if enemy is bigger or equal in length
          if (move.x === myHead.x && move.y === myHead.y + 1) {
            isMoveSafe.up = false;
          }
          if (move.x === myHead.x && move.y === myHead.y - 1) {
            isMoveSafe.down = false;
          }
          if (move.x === myHead.x - 1 && move.y === myHead.y) {
            isMoveSafe.left = false;
          }
          if (move.x === myHead.x + 1 && move.y === myHead.y) {
            isMoveSafe.right = false;
          }
        }
      });
    });
  
    return isMoveSafe;
}

export default avoidHeadToHead;
