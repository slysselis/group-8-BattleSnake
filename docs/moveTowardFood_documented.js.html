/**
 * Determines the best move to approach the closest food item
 * Uses Manhattan distance to find closest food and attempts to move toward it
 * @param {Object} gameState - The current game state from Battlesnake API
 * @param {Object} gameState.you - Your snake's data
 * @param {Array} gameState.you.body - Array of coordinate objects representing snake body
 * @param {Object} gameState.board - The game board data
 * @param {Array} gameState.board.food - Array of food coordinate objects
 * @param {Object} isMoveSafe - Object indicating which moves are currently safe
 * @param {boolean} isMoveSafe.up - Whether moving up is safe
 * @param {boolean} isMoveSafe.down - Whether moving down is safe
 * @param {boolean} isMoveSafe.left - Whether moving left is safe
 * @param {boolean} isMoveSafe.right - Whether moving right is safe
 * @returns {string|null} Direction to move toward food ("up", "down", "left", "right") or null if no safe food move
 * @example
 * // Try to move toward the closest food
 * const foodDirection = moveTowardFood(gameState, isMoveSafe);
 * if (foodDirection) {
 *   console.log(`Moving ${foodDirection} toward food`);
 * }
 */
function moveTowardFood(gameState, isMoveSafe) {
  const myHead = gameState.you.body[0];
  const food = gameState.board.food;

  // If there's food available, try to move toward the closest piece
  if (food.length > 0) {
    // Find the closest food using Manhattan distance
    let closestFood = food.reduce((closest, current) => {
      let closestDist = Math.abs(closest.x - myHead.x) + Math.abs(closest.y - myHead.y);
      let currentDist = Math.abs(current.x - myHead.x) + Math.abs(current.y - myHead.y);
      return currentDist < closestDist ? current : closest;
    });

    // Try to move toward the closest food if the move is safe
    // Prioritize horizontal movement first, then vertical
    if (closestFood.x < myHead.x && isMoveSafe.left) {
      return "left";
    }
    if (closestFood.x > myHead.x && isMoveSafe.right) {
      return "right";
    }
    if (closestFood.y < myHead.y && isMoveSafe.down) {
      return "down";
    }
    if (closestFood.y > myHead.y && isMoveSafe.up) {
      return "up";
    }
  }

  // If we couldn't move toward food safely, return null
  return null;
}

export default moveTowardFood;
