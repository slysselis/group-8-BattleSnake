/**
 * Performs flood fill algorithm to calculate available space from a starting position
 * @param {Array<Array<string>>} board - 2D array representing the game board ('.' for empty, 'X' for blocked)
 * @param {Object} start - Starting coordinate for flood fill
 * @param {number} start.x - X coordinate of starting position
 * @param {number} start.y - Y coordinate of starting position
 * @returns {number} Number of accessible empty spaces from the starting position
 * @example
 * // Calculate available space from position (5, 5)
 * const board = [
 *   ['.', '.', 'X'],
 *   ['.', '.', '.'],
 *   ['X', '.', '.']
 * ];
 * const space = floodFill(board, { x: 1, y: 1 });
 * console.log(space); // 5
 */
export function floodFill(board, start) {
    const visited = new Set();
    const queue = [start];
    const height = board.length;
    const width = board[0].length;
  
    // Movement directions: up, down, left, right
    const directions = [
      { x: 0, y: -1 }, { x: 0, y: 1 },
      { x: -1, y: 0 }, { x: 1, y: 0 }
    ];
  
    let space = 0;
  
    while (queue.length > 0) {
      const { x, y } = queue.pop();
      const key = `${x},${y}`;
  
      // Skip if out of bounds, already visited, or blocked
      if (
        x < 0 || y < 0 ||
        x >= width || y >= height ||
        visited.has(key) ||
        board[y][x] === 'X'
      ) continue;
  
      visited.add(key);
      space++;
  
      // Add adjacent cells to queue
      directions.forEach(dir => {
        queue.push({ x: x + dir.x, y: y + dir.y });
      });
    }
  
    return space;
}
